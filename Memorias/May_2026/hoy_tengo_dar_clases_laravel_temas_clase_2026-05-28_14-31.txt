### 3.2 Transacciones Complejas (15 min)
```php
class EnrollmentService
{
    /**
     * Inscribir estudiante en múltiples materias con validación
     * y manejo de stock de cupos
     */
    public function enrollStudentInSubjects(Student $student, array $subjectIds): void
    {
        DB::transaction(function () use ($student, $subjectIds) {
            // 1. Verificar disponibilidad de cupos
            $subjects = Subject::whereIn('id', $subjectIds)
                ->lockForUpdate() // ✅ Bloqueo pesimista
                ->get();
            
            foreach ($subjects as $subject) {
                if ($subject->enrollments()->count() >= $subject->capacity) {
                    throw new \Exception("{$subject->name} está llena.");
                }
            }
            
            // 2. Crear inscripciones
            $student->subjects()->attach($subjectIds, [
                'enrolled_at' => now(),
            ]);
            
            // 3. Actualizar estadísticas del aula
            $student->classroom->increment('active_enrollments');
            
            // 4. Registrar en log de auditoría
            ActivityLog::create([
                'action' => 'enrollment',
                'description' => "Estudiante {$student->name} inscrito en " . 
                                count($subjectIds) . " materias",
                'user_id' => auth()->id(),
            ]);
        }, 5); // Intentar 5 veces si hay deadlock
    }
    
    /**
     * Transferir estudiante de aula con rollback automático
     */
    public function transferClassroom(Student $student, Classroom $newClassroom): void
    {
        $oldClassroom = $student->classroom;
        
        DB::transaction(function () use ($student, $newClassroom, $oldClassroom) {
            // Guardar estado anterior por si se necesita restaurar
            $oldData = $student->toArray();
            
            // Actualizar estudiante
            $student->update(['classroom_id' => $newClassroom->id]);
            
            // Actualizar contadores
            if ($oldClassroom) {
                $oldClassroom->decrement('student_count');
                $oldClassroom->activities()->create([
                    'type' => 'student_left',
                    'student_id' => $student->id,
                    'transferred_to' => $newClassroom->id
                ]);
            }
            
            $newClassroom->increment('student_count');
            $newClassroom->activities()->create([
                'type' => 'student_arrived',
                'student_id' => $student->id,
                'transferred_from' => $oldClassroom->id ?? null
            ]);
            
            // Validaciones post-transacción
            if ($newClassroom->student_count > $newClassroom->capacity) {
                throw new \Exception("Aula {$newClassroom->name} excede capacidad");
            }
        });
    }
}
```

```php
{

            return response()->json([
                'error' => 'Error al procesar la solicitud',
                'code' => 'DATABASE_ERROR'
```

```php
namespace App\Http\Controllers\Api\Auth;

class AuthController extends Controller
{
    /**
     * Login con múltiples opciones de seguridad
     */
    public function login(LoginRequest $request): JsonResponse
    {
        // Validar rate limiting
        $key = 'login:' . $request->ip();
        $attempts = Cache::increment($key);
        
        if ($attempts > 5) {
            throw new \Exception('Demasiados intentos. Espera 1 minuto.');
        }
        Cache::expire($key, 60);
        
        // Autenticar usuario
        if (!Auth::attempt($request->only('email', 'password'), $request->remember)) {
            Log::warning('Login fallido', [
                'email' => $request->email,
                'ip' => $request->ip(),
                'user_agent' => $request->userAgent()
            ]);
            
            return response()->json([
                'message' => 'Credenciales inválidas'
            ], 401);
        }
        
        $user = Auth::user();
        
        // Verificar si el usuario está activo
        if (!$user->is_active) {
            Auth::logout();
            return response()->json([
                'message' => 'Cuenta desactivada'
            ], 403);
        }
        
        // Revocar tokens anteriores si es necesario
        if ($request->revoke_previous) {
            $user->tokens()->delete();
        }
        
        // Crear token con capacidades específicas
        $abilities = match($user->role) {
            'admin' => ['*'],
            'teacher' => ['students:read', 'students:write', 'grades:write'],
            'student' => ['profile:read', 'grades:read'],
            default => ['profile:read']
        };
        
        $token = $user->createToken(
            name: $request->device_name ?? 'web',
            abilities: $abilities,
            expiresAt: now()->addHours(24)
        );
        
        return response()->json([
            'user' => new UserResource($user),
            'token' => $token->plainTextToken,
            'abilities' => $abilities,
            'expires_at' => $token->accessToken->expires_at
        ]);
    }
    
    /**
     * Logout con limpieza completa
     */
    public function logout(Request $request): JsonResponse
    {
        $user = $request->user();
        
        // Opciones de logout
        if ($request->all_devices) {
            // Revocar todos los tokens
            $user->tokens()->delete();
            
            Log::info('Logout completo', ['user_id' => $user->id]);
        } else {
            // Revocar solo el token actual
            $request->user()->currentAccessToken()->delete();
        }
        
        return response()->json([
            'message' => 'Sesión cerrada exitosamente'
        ]);
    }
}
```

### 4.3 CRUD Completo con Middleware (15 min)
```php
namespace App\Http\Controllers\Api;

use App\Http\Requests\StoreStudentRequest;
use App\Http\Requests\UpdateStudentRequest;
use App\Http\Resources\StudentResource;
use App\Http\Resources\StudentCollection;

class StudentController extends Controller
{
    public function __construct()
    {
        // Middleware específico para cada método
        $this->middleware('ability:students:read')->only(['index', 'show']);
        $this->middleware('ability:students:write')->only(['store', 'update', 'destroy']);
        
        // Rate limiting por método
        $this->middleware('throttle:60,1')->only('index');
        $this->middleware('throttle:20,1')->only('store', 'update', 'destroy');
    }
    
    /**
     * @OA\Get(
     *     path="/api/students",
     *     summary="Listar estudiantes con filtros y paginación"
     * )
     */
    public function index(StudentIndexRequest $request): StudentCollection
    {
        $query = Student::query()
            ->with(['classroom:id,name,grade', 'subjects:id,name'])
            ->withCount('enrollments');
        
        // Filtros dinámicos
        $filters = $request->validated();
        
        $query->when($filters['search'] ?? null, function ($q, $search) {
            $q->where(function ($subQ) use ($search) {
                $subQ->where('name', 'like', "%{$search}%")
                     ->orWhere('email', 'like', "%{$search}%");
            });
        });
        
        $query->when($filters['classroom_id'] ?? null, function ($q, $id) {
            $q->where('classroom_id', $id);
        });
        
        $query->when($filters['grade_min'] ?? null, function ($q, $grade) {
            $q->whereHas('subjects', function ($subQ) use ($grade) {
                $subQ->where('grade', '>=', $grade);
            });
        });
        
        $query->when($filters['sort'] ?? null, function ($q, $sort) {
            $direction = $filters['direction'] ?? 'asc';
            $q->orderBy($sort, $direction);
        });
        
        // Cache de resultados
        $cacheKey = 'students:' . md5(serialize($filters) . $request->page);
        
        $students = Cache::remember($cacheKey, 300, function () use ($query) {
            return $query->paginate($request->per_page ?? 15);
        });
        
        return new StudentCollection($students);
    }
    
    /**
     * Crear estudiante con transacción y eventos
     */
    public function store(StoreStudentRequest $request): JsonResponse
    {
        $student = DB::transaction(function () use ($request) {
            // Crear estudiante
            $student = Student::create($request->safe()->except('subjects'));
            
            // Inscribir en materias si se especifica
            if ($subjects = $request->input('subjects')) {
                app(EnrollmentService::class)
                    ->enrollStudentInSubjects($student, $subjects);
            }
            
            // Disparar evento
            event(new StudentCreated($student));
            
            // Log de auditoría
            activity()
                ->performedOn($student)
                ->causedBy(auth()->user())
                ->withProperties(['subjects' => $subjects ?? []])
                ->log('Estudiante creado');
            
            return $student->load(['classroom', 'subjects']);
        }, 3); // 3 reintentos en caso de deadlock
        
        return response()->json([
            'message' => 'Estudiante creado exitosamente',
            'data' => new StudentResource($student)
        ], 201);
    }
    
    /**
     * Mostrar estudiante con todas sus relaciones
     */
    public function show(int $id): StudentResource
    {
        $student = Student::with([
            'classroom' => function ($q) {
                $q->select('id', 'name', 'grade', 'capacity');
            },
            'enrollments' => function ($q) {
                $q->with(['subject' => function ($sq) {
                    $sq->select('id', 'name', 'credits');
                }])->latest();
            },
            'enrollments.teacher',
            'activities' => function ($q) {
                $q->latest()->limit(10);
            }
        ])->findOrFail($id);
        
        return new StudentResource($student);
    }
    
    /**
     * Actualizar con validación y transacción
     */
    public function update(UpdateStudentRequest $request, int $id): JsonResponse
    {
        $student = Student::findOrFail($id);
        
        DB::transaction(function () use ($student, $request) {
            $oldValues = $student->toArray();
            $student->update($request->safe()->all());
            
            // Si cambia de aula, actualizar contadores
            if ($student->wasChanged('classroom_id')) {
                Classroom::where('id', $oldValues['classroom_id'])
                    ->decrement('student_count');
                Classroom::where('id', $student->classroom_id)
                    ->increment('student_count');
            }
            
            // Log de cambios
            activity()
                ->performedOn($student)
                ->withProperties([
                    'old' => $oldValues,
                    'new' => $student->toArray()
                ])
                ->log('Estudiante actualizado');
        });
        
        return response()->json([
            'message' => 'Estudiante actualizado',
            'data' => new StudentResource($student->fresh())
        ]);
    }
    
    /**
     * Eliminar con soft delete y cascada
     */
    public function destroy(int $id): JsonResponse
    {
        $student = Student::findOrFail($id);
        
        DB::transaction(function () use ($student) {
            // Eliminar inscripciones
            $student->enrollments()->delete();
            
            // Actualizar contador del aula
            Classroom::where('id', $student->classroom_id)
                ->decrement('student_count');
            
            // Soft delete del estudiante
            $student->delete();
            
            // Log
            activity()
                ->performedOn($student)
                ->log('Estudiante eliminado');
        });
        
        return response()->json([
            'message' => 'Estudiante eliminado correctamente'
        ]);
    }
}
```

```php
// StudentResource.php
class StudentResource extends JsonResource
{
    public function toArray(Request $request): array
    {
        $user = $request->user();
        
        return [
            'id' => $this->id,
            'name' => $this->name,
            'email' => $this->email,
            'phone' => $this->phone,
            
            // Relaciones condicionales
            'classroom' => ClassroomResource::make($this->whenLoaded('classroom')),
            'subjects' => SubjectResource::collection($this->whenLoaded('subjects')),
```

```php
class CacheManager
{
    /**
     * Sistema de cache multi-nivel para estudiantes
     */
    public function getStudentsList(array $filters): array
    {
        $cacheKey = 'students:' . md5(serialize($filters));
        $tags = ['students', 'students_index'];
        
        // ✅ Cache en memoria (Redis)
        if ($cached = Cache::tags($tags)->get($cacheKey)) {
            return $cached;
        }
        
        // ✅ Cache en archivo (segundo nivel)
        if (config('cache.default') === 'file') {
            return $this->fileCacheSystem($cacheKey, $filters);
        }
        
        // ✅ Generar y cachear
        $data = Student::with(['classroom:id,name'])
            ->filter($filters)
            ->paginate(20)
            ->toArray();
        
        Cache::tags($tags)->put($cacheKey, $data, now()->addHours(1));
        
        return $data;
    }
    
    /**
     * Invalidación inteligente de cache
     */
    public function invalidateStudentCache(Student $student): void
    {
        // Eliminar cache relacionado con este estudiante
        Cache::tags([
            "student:{$student->id}",
            'students_index'
        ])->flush();
        
        // También invalidar cache del aula
        if ($student->classroom_id) {
            Cache::tags([
                "classroom:{$student->classroom_id}",
                'classrooms_index'
            ])->flush();
        }
        
        // Notificar a otros servicios
        Cache::add('cache:invalidated:students', now(), 10);
    }
}
```
